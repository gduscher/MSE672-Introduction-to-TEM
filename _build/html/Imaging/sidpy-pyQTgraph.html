"""
Demonstrates selecting plot curves by mouse click
"""

import numpy as np
import pyTEMlib.file_tools as ft
from pyqtgraph.jupyter import GraphicsLayoutWidget
import pyqtgraph as pg

datasets = ft.open_file()
eels_dataset = datasets['Channel_000']
print(eels_dataset.data_type.name)
print(eels_dataset)

pg.mkQApp()
win = GraphicsLayoutWidget(css_width="800px", css_height="480px")

p1 = win.addPlot(title="sidpy plot")
vb = p1.getViewBox()
#vb.setBackgroundColor('white')

spectrum = pg.PlotCurveItem(x = eels_dataset.energy_loss, y=np.array(eels_dataset)[0,0,:-1], stepMode=True, fillLevel=0, padding = 0, pen='b', clickable=True)
p1.addItem(spectrum)


display(win)

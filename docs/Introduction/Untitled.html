import numpy as np
from matplotlib import cbook
import matplotlib.colors
import matplotlib.pyplot as plt
from mpl_toolkits.axes_grid1 import ImageGrid

# from https://matplotlib.org/3.3.3/gallery/axes_grid1/demo_axes_grid2.html
# *** Demo 2: shared colorbar ***
plt.rcParams["mpl_toolkits.legacy_colorbar"] = False

fig = plt.figure()
grid2 = ImageGrid(fig, 111,
                  nrows_ncols=(2, 2),
                  axes_pad=0.35,
                  label_mode="L",
                  share_all=True,
                  cbar_location="right",
                  cbar_mode="single",
                  cbar_size="4%",
                  cbar_pad=0.05,
                  )

grid2[0].set_xlabel("X")
grid2[0].set_ylabel("Y")

ZS = np.random.random([4,10,10])
vmax, vmin = np.max(ZS), np.min(ZS)
norm = matplotlib.colors.Normalize(vmax=vmax, vmin=vmin)


for ax, z in zip(grid2, ZS):
    im = ax.imshow(z, norm=norm, origin="lower")#, extent=extent)

# With cbar_mode="single", cax attribute of all axes are identical.
    ax.cax.colorbar(im)
